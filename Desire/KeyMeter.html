def delete(subkey)
  bytes = key_meter.zscore(subkey).to_i

  key_meter.prepare do
    key_meter.zrem(subkey)
    key_meter.zincrby(-bytes, @total_key)
  end
  storage_meter.incrby(-bytes)
end

def update(subkey, bytes)
  key_meter.prepare do
    key_meter.zadd(bytes, subkey)
    key_meter.zincrby(bytes, @total_key)
  end
  storage_meter.incrby(bytes)
end
