def initialize(client, klass, base_key, duration=:day)
  @base_key = base_key
  @client = client
  @collector = Collector.new(client, base_key) do |subkey|
    klass.new(@client, subkey)
  end
  case duration
  when :month
    @format_string = '%Y-%m'
  when :day
    @format_string = '%Y-%m-%d'
  when :hour
    @format_string = '%Y-%m-%dT%H'
  else
    raise ArgumentError, "Unsupported duration: #{duration}"
  end
end
